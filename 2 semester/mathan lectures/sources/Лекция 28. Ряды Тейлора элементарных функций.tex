\section{Ряды Тейлора элементарных функций}

\begin{theorem}
    Функции $e^x, \cos x, \sin x, \ch x, \sh x$ представимы своим рядом Тейлора с центром в нуле (рядом Маклорена) на всей $\mathbb{R}$, то есть
    \[ e^x = \sum_{k=0}^{\infty} \frac{x^k}{k!} \ \forall x\in \mathbb{R} \]   
    \[ \ch x = \sum_{k=0}^{\infty} \frac{x^{2k}}{(2k)!} \ \forall x\in \mathbb{R} \]
    \[ \sh x = \sum_{k=0}^{\infty} \frac{x^{2k+1}}{(2k+1)!} \forall x\in \mathbb{R} \]
    \[ \sin x = \sum_{k=0}^{\infty} \frac{(-1)^k x^{2k+1}}{(2k+1)!} \ \forall x\in \mathbb{R} \]   
    \[ \cos x = \sum_{k=0}^{\infty} \frac{(-1)^k x^{2k}}{(2k)!} \ \forall x\in \mathbb{R} \]
\end{theorem}


\begin{proof}
    Фиксируем произвольный $\delta>0$. 
    \[ f(x) = e^x \]
    \[ f^{(k)} (x) = e^x \]  
    \[ \exists M(\delta)= e^\delta > 0: \forall x\in (-\delta,\delta) \hookrightarrow | f^{(k)}(x) | \le e^\delta \]   
    В силу достаточного условия регулярности ряд Тейлора $e^x$ сходится к $e^x \ \forall x\in (-\delta,\delta).$ В сильно произвольности $\delta>0$ получаем сходимость ряда Тейлора $\forall x\in \mathbb{R}$.
    Аналогичные рассуждения применимы и к $ \cos x, \sin x, \ch x, \sh x$.
\end{proof}

\begin{remark}
    Напомним, что 
    \[ e^z := e^x (\cos y+i\sin y) \text{, если  } z= x+iy \]
\end{remark}

\begin{theorem}
    (О комплексной экспоненте) 
    \[ \forall z\in \mathbb{C} \hookrightarrow e^z = \sum_{k=0}^{\infty} \frac{z^k}{k!} = f(z) \]
\end{theorem}

\begin{proof}
    1) Мы уже знаем, что $\displaystyle \sum_{k=0}^{\infty} \frac{x^k}{k!}$ сходится $\forall x\in \mathbb{R}$ к $e^x$.
    В силу теоремы о круге сходимости ряд $\displaystyle \sum_{k=0}^{\infty} \frac{z^k}{k!}$ сходится абсолютно внутри круга сходимости и $k$-ый член не стремится к 0 вне круга сходимости, но у нас есть сходимость $\forall z\in \mathbb{R}$. Такое может 
    быть только в том случае, если $R_{cx} = \infty \Rightarrow$ степенной ряд $\displaystyle \sum_{k=0}^{\infty} \frac{z^k}{k!}$  сходится абсолютно $\forall z\in \mathbb{C}$.\\
    2) Покажем, что $\displaystyle f(z) : = \sum_{k=0}^{\infty} \frac{z^k}{k!} \ \forall z\in \mathbb{C}$ удовлетворяет функциональному уравнению
    \[ f(z_1+z_2) = f(z_1)f(z_2) \ \forall z_1,z_2\in \mathbb{C} \]
    Теорема о перемножении абсолютно сходящихся рядов справедлива и в комплексном случае (доказательство аналогично вещественному случаю)
    \[ f(z_1)f(z_2) = ( \sum_{k=0}^{\infty}\frac{z_1^k}{k!} ) (\sum_{k=0}^{\infty}\frac{z_2^k}{k!}  ) = \] 
    Введём произвольную биекцию. Пусть $(n,m): \mathbb{N} \to \mathbb{N}^2 $ \\
    $(n(j),m(j))$ --- пара натуральных чисел 
    \[ = \sum_{j=1}^{\infty} \frac{z_1^{n(j)}z_2^{m(j)}}{n(j)!m(j)!} \]   
      \begin{figure}[H]
        \centering
        \includegraphics[width=0.35\textwidth]{pictures/произвольная биекция.png}
        
    \end{figure}

    При таком способе нумерации 
    \[ \sum_{j=1}^{\infty} \frac{z_1^{n(j)}z_2^{m(j)}}{n(j)!m(j)!} = \sum_{s=0}^{\infty} \frac{1}{s!} \sum_{k=0}^{s} s! \frac{z_1^k z_2^{s-k}}{k! (s-k)!} = \sum_{s=0}^{\infty} \frac{1}{s!} (z_1+z_2)^s= f(z_1+z_2) \]
    3) \[ f(iy) = \sum_{k=0}^{\infty} \frac{(iy)^k}{k!} = \sum_{\substack{k=0 \\ \text{k-четное}}}^{\infty} \frac{(iy)^k}{k!} + \sum_{\substack{k=0 \\ \text{k-нечетное}}}^{\infty} \frac{(iy)^k}{k!} = \sum_{k=0}^{\infty} \frac{(-1)^k y^{2k}}{(2k)!} + i\sum_{k=0}^{\infty} \frac{(-1)^k y^{2k+1}}{(2k+1)!} = \cos y+i\sin y \ \forall y\in \mathbb{R} \] 
    4) $\forall x\in \mathbb{R} \ f(x) = e^x$ (доказали в предыдущей теореме) \\
    $\forall y\in \mathbb{R} \ f(iy) = \cos y+ i\sin y$ (на 3 шаге)\\
    Но из 2) шага $\Rightarrow f(x+iy) = f(x)f(iy) = e^x(\cos y+i\sin y)$ 
\end{proof}

\begin{definition}
    \[ \ch z:= \sum_{k=0}^{\infty} \frac{z^{2k}}{(2k)!} \ \forall z\in \mathbb{C} \]   
    \[ \sh z := \sum_{k=0}^{\infty} \frac{z^{2k+1}}{(2k+1)!} \ \forall z\in \mathbb{C} \]   
    \[ \sin z:= \sum_{k=0}^{\infty} \frac{(-1)^k z^{2k+1}}{(2k+1)!} \ \forall z\in \mathbb{C} \] 
    \[ \cos z:= \sum_{k=0}^{\infty} \frac{(-1)^k z^{2k}}{(2k)!} \ \forall z\in \mathbb{C} \]   
\end{definition}


\begin{remark}
    Это определение корректно, то есть $R_{cx} = +\infty$. Доказательство такое же, как и в предудыщей теореме.
\end{remark}

\begin{consequence}
    (Формулы Эйлера)
    \[ e^{iz} = \cos z+i\sin z \ \forall z\in \mathbb{C}\]
    \[ \cos z = \frac{e^{iz}+e^{-iz}}{2} \ \forall z\in \mathbb{C} \] 
    \[ \sin z= \frac{e^{iz}-e^{-iz}}{2i} \ \forall z\in \mathbb{C} \]
\end{consequence}

\begin{proof}
    Докажем первую формулу. По предыдущей теореме:
    \[ e^{iz} = \sum_{k=0}^{\infty} \frac{(iz)^k}{k!} \ \forall z\in \mathbb{C} \]   
    Но 
    \[ \sum_{k=0}^{\infty} \frac{(iz)^k}{k!} = \sum_{\substack{k=0 \\ \text{k-четное}}}^{\infty} \frac{(-1)^k z^{2k}}{(2k)!} +  i\sum_{\substack{k=0 \\ \text{k-нечетное}}}^{\infty} \frac{(-1)^k z^{2k+1}}{(2k+1)!} = \cos z+i\sin z \text{ по предыдущему определению}\]
\end{proof}

\begin{theorem}
    (формула Тейлора с остаточным членом в интегральной форме). Пусть $f\in C^{n+1} ((\underline{x}-\delta, \underline{x}+\delta)), \delta>0 \, \underline{x}\in \mathbb{R}$. Тогда
    \[ f(x) = T_{\underline{x}}^n [f](x) + \frac{1}{n!} \int_{\underline{x}}^{x} (x-t)^n f^{(n+1)}(t)\, dt \ \forall x\in (\underline{x}-\delta, \underline{x}+\delta) \]
\end{theorem}

\begin{proof}
    Докажем по индукции.\\
    При $n=0 \ T_{\underline{x}}^n [f](x) = f(\underline{x})$\\
    \[ f(x) -f(\underline{x}) = \int_{\underline{x}}^{x} f'(t)\, dt \text{ формула Ньютона-Лейбница} \] 
    Пусть доказано при $n=s$. Покажем, что верно при $s+1$
    \[ f(x) = \sum_{k=0}^{s} \frac{f^{(k)}(\underline{x})}{k!} (x-\underline{x})^k + \frac{1}{s!} \int_{\underline{x}}^{x} (x-t)^s f^{(s+1)}(t)\,dt \] 
    \[ \int_{\underline{x}}^{x} (x-t)^s f^{(s+1)}(t)\,dt = \text{ по частям } = - \frac{(x-t)^{s+1}}{s+1}f^{(s+1)}(t) \bigg|_{\underline{x}}^{x} + \int_{\underline{x}}^{x} \frac{(x-t)^{s+1}}{(s+1)!} f^{(s+2)} (t)\,dt = \]
    \[ = \frac{(x-\underline{x})^{s+1}f^{(s+1)}(\underline{x})}{s+1} + \int_{\underline{x}}^{x} \frac{(x-t)^{s+1}}{(s+1)}f^{(s+2)}(t)\,dt \] 
    \[ \Rightarrow f(x) = \sum_{k=0}^{s+1} \frac{(x-\underline{x})^k f^{(k)}(\underline{x})}{k!} + \frac{1}{(s+1)!} \int_{\underline{x}}^{x} (x-t)^{s+1} f^{(s+2)}(t)\,dt \]
    $\Rightarrow$ индукционный шаг сделан.
\end{proof}

\begin{theorem}
    \[ \forall \alpha\in \mathbb{R} \ (1+x)^\alpha = \sum_{k=0}^{\infty} C_\alpha ^k x^k \ \forall x\in (-1,1) \] 
\end{theorem}

\begin{proof}
    Если $\alpha\in \mathbb{N} \cup \{0\}$, то формула вырождается в бином Ньютона (является вырожденным случаем ряда Тейлора, так как коэффициенты при $n>\alpha =0)$. Далее считаем, что $\alpha\notin \mathbb{N} \cup \{0\}$.
    В прошлом семестре мы уже провели вычисления и доказали, что если $f(x) = (1+x)^\alpha$, то $\displaystyle \frac{f^{(k)}(0)}{k!} = C_\alpha^k = \frac{\alpha \cdot (\alpha-1) \cdot ... \cdot (\alpha - k +1 )}{k!} $.
    Покажем, что 
    \[ (1+x)^\alpha - \sum_{k=0}^{n} C_\alpha^k x^k \to 0, n\to \infty \ \forall x\in (-1,1) \]
    Воспользуемся формулой Тейлора с остаточным членом в интегральной форме: 
    \[ (1+x)^\alpha - \sum_{k=0}^{n} C_\alpha^k x^k = \frac{1}{n!} \int_{0}^{x} (x-t)^n f^{(n+1)}(t)\,dt \] 
    \[ f^{(n+1)} (t) = \alpha (\alpha-1) \cdot ... \cdot (\alpha-n ) \cdot (1+t)^{\alpha-n-1} \]
    Тогда 
    \[ (1+x)^\alpha - \sum_{k=0}^{n} C_\alpha^k x^k = \frac{1}{n!} \int_{0}^{x} (x-t)^n f^{(n+1)}(t)\,dt = \frac{\alpha(\alpha-1)\cdot ... \cdot (\alpha-n)}{n!} \int_{0}^{x} (x-t)^n (1+t)^{\alpha-n-1}\,dt = \] 
    Сделаем хитрую замену $t=\tau x$ и пусть $\displaystyle \lambda_n = \frac{\alpha (\alpha-1) \cdot ... \cdot (\alpha-n )}{n!}.$ 
    Тогда это всё будет равно 
    \[ = \lambda_n \int_{0}^{1} (x-\tau x)^n (1+\tau x)^{\alpha-n-1} x \,d\tau = \lambda_n x^{n+1} \int_{0}^{1}  \frac{(1-\tau)^n}{(1+\tau x)^n} (1+\tau x)^{\alpha-1}\, d\tau  \]
    \[ 0 \le \frac{(1-\tau)^n}{(1+\tau x)^n} \le 1 \text{ так как } 1-\tau \le 1+\tau x \text{ так как } x\in (-1,1) \]
    \[ \left| \lambda_n \int_{0}^{1} (x-\tau x)^n (1+\tau x)^{\alpha-n-1} x \,d\tau \right| \le \lambda_n |x|^{n+1} \left| \int_{0}^{1} (1+\tau x)^{\alpha-1} \,d\tau \right| \]
    Пусть $\displaystyle \int_{0}^{1} (1+\tau x)^{\alpha-1} \,d\tau$ это какое-то число C, которое на зависит от n.
    Будем считать, что $x\neq 0$, так как при x=0 всё тривиально.
    \[ \left| \frac{\lambda_{n+1} |x|^{n+2}}{\lambda_n |x|^{n+1}} \right| = \left| |x| \frac{(\alpha-n-1)}{n+1} \right|  \to | -|x| | < 1, n\to \infty \]   
    Выберем число $q\in (|x|,1).$Тогда 
    \[ \Rightarrow \exists N(q)\in \mathbb{N}: \forall n\ge N(q) \ \left| \frac{\lambda_{n+1}x^{n+2}}{\lambda_nx^{n+1}} \right| \le q \]
    \[ | \lambda_{N(q)+m} x^{N(q)+m+1} | \le q^m | \lambda_{N(q)} x^{N(q)} | \to 0, m\to \infty  \]   
    \[ \Rightarrow \lambda_n|x|^{n+1}\to 0, n\to \infty \] 
    А значит,
    \[ \lambda_n |x|^{n+1} \left| \int_{0}^{1} (1+\tau x)^{\alpha-1} \,d\tau \right| \to 0, n\to \infty \]

\end{proof}


\begin{remark}
    $R_{cx}$ ряда $\displaystyle \sum_{k=0}^{\infty} C_\alpha^k x^k =1 .$
\end{remark}


\begin{proof}
    \[ x^{km+n} \quad n=0, m=1 \]  
    \[ \lim\limits_{k\to \infty} \left| \frac{C_\alpha^{k+1}}{C_\alpha^k} \right| = \left| \frac{\alpha \cdot ... \cdot (\alpha-k)}{(k+1)!} \cdot \frac{k!}{\alpha\cdot ... \cdot(\alpha-k+1)} \right| = \left| \frac{\alpha-k}{k+1}\right| \to 1, k\to \infty \]
\end{proof}


\begin{remark}
    \[ (1+x)^{-1} = \sum_{k=0}^{\infty} (-1)^k x^k, x\in (-1,1) \] 
    \[ \Rightarrow \ln (1+x) = \sum_{k=0}^{\infty} \frac{(-1)^k x^{k+1}}{k+1} \forall x\in (-1,1) \text{ по теореме о почленном интегрировании степенного ряда  } \] 
    в интервале сходимости
    Или переобозначив индексы можно записать это более красиво:
    \[ \ln (1+x) = \sum_{k=1}^{\infty} \frac{(-1)^{k-1}x^k}{k} \]
    Интересное наблюдение: на самом деле, $\displaystyle \ln 2= \sum_{k=1}^{\infty} \frac{(-1)^{k-1}}{k}$
\end{remark}

\begin{proof}
    Для обоснования интересного равенства $\displaystyle \ln 2= \sum_{k=1}^{\infty} \frac{(-1)^{k-1}}{k}$ заметим, что правая часть сходится по признаку Лейбница $\Rightarrow$ степенной ряд $\displaystyle \sum_{k=1}^{\infty} \frac{(-1)^{k-1}x^k}{k}$ сходится при $x=1\Rightarrow$ по второй теореме Абеля $\hookrightarrow$ ряд сходится равномерно на $[0,1]$,
    но на интервале $(0,1)$ он сходится к $\ln (1+x)$, который непрерывен на $[0,1]$. Из свойств равномерно сходящихся рядов $\displaystyle \left( \sum_{k=1}^{\infty} \frac{(-1)^{k-1} x^{k}}{k} \right) \bigg|_{x=1}= \sum_{k=1}^{\infty} \frac{(-1)^k 1}{k}$. Сделаем предельный переход 
    и получим, что $\displaystyle \left( \sum_{k=1}^{\infty} \frac{(-1)^{k-1} x^{k}}{k} \right) = \ln 2$,так как сумма ряда непрерывна на $[0,1]$.
\end{proof}


\begin{remark}
    Запишем ряд для $\arctg$
    \[ (1+t^2)^{-1} = \sum_{k=0}^{\infty} (-1)^k t^{2k} \] 
    \[ \Rightarrow \arctg t = \sum_{k=0}^{\infty} \frac{(-1)^k t^{2k+1}}{2k+1} \ \forall t\in (-1,1) \]   
    Запишем еще напоследок для $\arcsin$
    \[ (1+t)^{-\frac{1}{2}} = \sum_{k=0}^{\infty} \frac{(-\frac{1}{2}) \cdot ... \cdot (-\frac{1}{2} -k+1) t^k }{k!} = \sum_{k=0}^{\infty} \frac{(-1)^k (2k-1)!!t^k}{2^k k!}   \]
    По определению считаем, что $(-1)!! = 1$
    \[ \arcsin t = \sum_{k=0}^{\infty} \frac{ t^{2k+1} (2k-1)!!}{2^k k! (2k+1)} \ \forall t\in (-1,1) \]
\end{remark}
