\section{Определённый интеграл Римана}

\begin{definition}
    \textbf{Разбиением} $T$ отрезка $[a, b]$ назовём конечный упорядоченный по 
    возрастанию набор точек, то есть 
    \[T = \{x_i\}^{N_T}_{i = 0}, \; \; N_T \in \mathbb{N}\]
    \[a = x_0 < x_1 < \dots < x_{N_T} = b\]
    $\mathcal{T}([a, b])$ -- множество всех разбиений отрезка $[a, b]$
\end{definition}

\begin{definition}
    Если $T \in \mathcal{T}[a, b]$, то \textbf{мелкость} разбиения 
    \[l(T) := \max_{1 \leq i \leq N_T} \left\lvert x_i - x_{i - 1} \right\rvert \]
\end{definition}

\begin{definition}
    Пусть $T \in \mathcal{T}([a, b])$. Пусть $f \colon [a, b] \to \mathbb{R}$. Тогда
    \[s_f(T) := \sum^{N_T}_{i = 1} m_i(f)(x_i - x_{i - 1}) \text { --- \textbf{нижняя сумма Дарбу}}\]
    \[S_f(T) := \sum^{N_T}_{i = 1} M_i(f)(x_i - x_{i - 1}) \text { --- \textbf{верхняя сумма Дарбу}}\] \\
    Здесь также определим $m_i(f)$ и $M_i(f)$
    \[m_i(f) := \inf_{x \in [x_{i - 1}, x_i]}f(x)\]
    \[M_i(f) := \sup_{x \in [x_{i - 1}, x_i]}f(x)\]
\end{definition}

\begin{remark}
    При этом будем придерживаться следующих соглашений:
    \begin{enumerate}
        \item $(+\infty) + (+\infty) = +\infty$
        \item $(-\infty) + (-\infty) = -\infty$
        \item $\lambda(\pm \infty) = \pm \infty \text{, если } \lambda > 0$
        \item $\lambda(\pm \infty) = \mp \infty \text{, если } \lambda < 0$
    \end{enumerate}
\end{remark}

\begin{remark}
    Для лучшего понимания сумм Дарбу можно рисовать себе в голове картинку
    \begin{figure}[H]
        \centering
        \includegraphics[width=0.8\linewidth]{pictures/Суммы Дарбу.png}
    \end{figure} \leavevmode
    \\
    Нижняя сумма Дарбу -- площадь зелёной части. \\
    Верхняя сумма Дарбу -- площадь зелёной и серой частей.
\end{remark}

\begin{lemma}
    Пусть $f \colon [a, b] \to \mathbb{R}$. Тогда 
    \begin{enumerate}
        \item $s_f(T) > -\infty \quad \forall T \in \mathcal{T}([a, b]) \ 
        \Longleftrightarrow \text{ f ограничена снизу на } [a, b]$
        \item $S_f(T) < +\infty \quad \forall T \in \mathcal{T}([a, b]) \ 
        \Longleftrightarrow \text{ f ограничена сверху на } [a, b]$
    \end{enumerate}
\end{lemma}

\begin{proof}
    Докажем первое, так как второе доказывается аналогично: \\
    $(\Longleftarrow):$ Если $f$ ограничена снизу на $[a, b]$, то 
    \[\forall T \in \mathcal{T}([a, b]) \hookrightarrow m_i(f) > -\infty 
    \quad \forall i \in \{1, \dots, N_T\}\]
    Следовательно, $s_f(T) > -\infty$. \\ 
    $(\Longrightarrow):$ Пусть 
    \[\forall T \in \mathcal{T}([a, b]) \hookrightarrow s_f(T) > -\infty\]
    Фиксируем произвольное разбиение $T = \{x_i\}^{N_T}_{i = 0}$. Тогда: 
    \[\forall i \in \{1, \dots, N_T\} \hookrightarrow m_i(f) > -\infty\]
    Воспользовавшись определением $m_i(f)$, получим 
    \[\forall i \in \{1, \dots, N_T\} \hookrightarrow \inf_{x \in [x_{i - 1}, x_i]}f(x) > -\infty\]
    Но 
    \[\displaystyle \bigcup^{N_T}_{i = 1}[x_{i - 1}, x_i] = [a, b]\]
    Следовательно,
    \[\inf_{x \in [a, b]} f(x) \geq 
    \min_{1 \leq i \leq N_T} m_i(f) > - \infty \implies f \text{ -- ограничена снизу на } [a, b]\]
\end{proof}

\begin{definition}
    Функция $f \colon [a, b] \to \mathbb{R}$ называется \textbf{интегрируемой по Риману} на $[a, b]$ 
    и обозначается $f \in \mathcal{R}([a, b])$, если 
    \[\exists J \in \mathbb{R} : 
    \begin{cases}
        \exists \displaystyle \lim_{l(T) \to 0} s_f(T) = J \\
        \exists \displaystyle \lim_{l(T) \to 0} S_f(T) = J  
    \end{cases}\]
    \[\Longleftrightarrow 
    \begin{cases}
        \forall \varepsilon > 0 \ \exists \delta(\varepsilon) > 0: \ \forall T \in \mathcal{T}([a, b]): \ 
        l(T) < \delta(\varepsilon) \ \hookrightarrow \ \left\lvert s_f(T) - J \right\rvert < \varepsilon \\
        \forall \varepsilon > 0 \ \exists \delta(\varepsilon) > 0: \ \forall T \in \mathcal{T}([a, b]): \ 
        l(T) < \delta(\varepsilon) \ \hookrightarrow \ \left\lvert S_f(T) - J \right\rvert < \varepsilon
    \end{cases}\]
    Если $f$ интегрируема на $[a, b]$ по Риману, то число $J$ называется \textbf{интегралом Римана} функции $f$ по отрезку 
    $[a, b]$ и обозначается \[\int_{a}^{b} f(x)\,dx \]
\end{definition}

\begin{definition}
    $\mathcal{R}([a, b]) \text{ --- класс всех функций, 
    интегрируемых по Риману на } [a, b]$.
\end{definition}

\section{Необходимое условие интегрируемости по Риману}

\begin{theorem}[Необходимое условие интегрируемости по Риману]
    Если $f \in \mathcal{R}([a, b])$, то $f$ ограниченна на $[a, b]$.
\end{theorem}

\begin{proof}
    Если $f \in \mathcal{R}([a, b])$, то $\exists J \in \mathbb{R}:$
    \[\forall \varepsilon > 0 \ \exists \delta(\varepsilon) > 0: \ \forall T 
    \in \mathcal{T}([a, b]): \ l(T) < \delta(\varepsilon) \hookrightarrow 
    \begin{cases}
        \left\lvert s_f - J \right\rvert < \varepsilon \\
        \left\lvert S_f - J \right\rvert < \varepsilon 
    \end{cases}\]
    Следовательно, подставляя $\varepsilon = 1$, получаем, что $\exists$ 
    разбиение $T \in \mathcal{T}([a, b])$ такое, что 
    \[\begin{cases}
        s_f(T) > - \infty \\
        S_f(T) < + \infty
    \end{cases}\]
    Рассуждая как при доказательстве леммы 1.1 ($\Longrightarrow$), получим, что 
    $f$ ограничена снизу и $f$ ограничена сверху на $[a, b] \implies f$ 
    ограничена на $[a, b]$.
\end{proof}

\begin{example}
    Рассмотрим функцию Дирихле на отрезке $[0, 1]$, то есть
    \[\mathcal{D}(x) = 
    \begin{cases}
        1, & x \in [0, 1] \cap \mathbb{Q} \\
        0, & x \in [0, 1] \cap (\mathbb{R} \backslash \mathbb{Q})
    \end{cases}\]
    Поймём, что $\mathcal{D}$ ограничена на $[0, 1]$, но $\mathcal{D} \notin \mathcal{R}([0, 1])$. \\
    Действительно, заметим, что 
    \[\begin{cases}
        s_{\mathcal{D}}(T) = 0 & \forall T \in \mathcal{T}([0, 1]) \\
        S_{\mathcal{D}}(T) = 1 & \forall T \in \mathcal{T}([0, 1])
    \end{cases}\]
    Теперь предположим, что $\mathcal{D} \in \mathcal{R}([0, 1])$. Тогда 
    \[\forall \varepsilon > 0 \ \exists \delta(\varepsilon) > 0 : \ \forall T \in 
    \mathcal{T}([0, 1]): \ l(T) < \delta(\varepsilon)\] 
    \[\hookrightarrow \ \left\lvert s_{\mathcal{D}}(T) - S_{\mathcal{D}}(T) \right\rvert \ \leq \ 
    \left\lvert s_{\mathcal{D}}(T) - J \right\rvert + \left\lvert S_{\mathcal{D}}(T) - J \right\rvert 
    \ < \ 2 \varepsilon\]
    Возьмём $\varepsilon = \displaystyle \frac{1}{2}$ и получим противоречие с нашим замечанием 
    о значениях $s_{\mathcal{D}}(T)$ и $S_{\mathcal{D}}(T)$.
\end{example}

\section{Первый критерий интегрируемости по Риману}

\begin{definition}
    Введём на множестве $\mathcal{T}([a, b])$ частичный порядок.
    Пусть $\ T_1, T_2 \in \mathcal{T}([a, b])$ такие, что 
    \[T_1 = \{x_i^1\}^{N_1}_{i = 0}
    \quad \text{ и } \quad T_2 = \{x_i^2\}^{N_2}_{i = 0}\]
    Тогда $\ T_1 \subset T_2 \ \Longleftrightarrow \ T_1 \preceq T_2$
\end{definition}

\begin{lemma}
    Пусть $f \colon [a, b] \to \mathbb{R}$. Пусть $T_1, T_2 \in 
    \mathcal{T}([a, b])$ такие, что $T_1 \preceq T_2$. Тогда 
    \begin{enumerate}
        \item $s_f(T_1) \leq s_f(T_2)$
        \item $S_f(T_1) \geq S_f(T_2)$
    \end{enumerate}
\end{lemma}

\begin{proof}
    Докажем первое, так как второе доказывается аналогично. \\
    Достаточно рассмотреть случай, когда $T_1$ и $T_2$ отличаются одной точкой 
    (в общем случае надо построить конечную упорядоченную цепочку разбиений, 
    отличающихся по одной точке). \\ \\
    Пусть $\ T_1 = \{x_i\}^N_{i = 0} \quad \quad a = x_0 < x_1 < \dots < x_N = b$ \\ \\
    $\exists j \in \{1, \dots, N\} \text{ и } \exists c \in (x_{j - 1}, x_j) \ 
    \text{ т.ч. } \ T_2 = T_1 \cup \{c\}$ \\ \\
    Случай $s_f(T_1) = - \infty$ тривиален, так как мы сразу же получаем верное неравенство. \\
    Тогда \[s_f(T_2) - s_f(T_1) \ = \ \inf_{x \in [x_{j - 1}, c]}f(x)(c - x_{j - 1}) \ + \ 
    \inf_{x \in [c, x_j]}f(x)(x_j - c) \ - \ \inf_{x \in [x_{j - 1}, x_j]}f(x)(x_j - x_{j - 1}) =\]
    \[= \ (c - x_{j - 1})\left[\inf_{x \in [x_{j - 1}, c]}f(x) \ - \ \inf_{x \in [x_{j - 1}, x_j]}f(x)\right] 
    \ + \ (x_j - c)\left[\inf_{x \in [c, x_j]}f(x) \ - \ \inf_{x \in [x_{j - 1}, x_j]}f(x)\right]\] \\
    Вспомним, что инфимум по меньшему множеству не может быть меньше инфимума по большему множеству. 
    Следовательно, каждая квадратная скобка будет неотрицательна (круглые скобки будут положительны, 
    так как $x_{j - 1} < c < x_j$). Заметим, что всё доказано.
\end{proof}

\begin{lemma}
    Пусть $f \colon [a, b] \to \mathbb{R}$. Пусть $T_1, T_2 \in 
    \mathcal{T}([a, b])$. Тогда $s_f(T_1) \leq S_f(T_2)$.
\end{lemma}

\begin{proof}
    $T_1 = \{x^1_i\}^{N_1}_{i = 0} \quad T_2 = \{x^2_i\}^{N_2}_{i = 0}$ \\ \\
    Определим $T$ -- разбиение отрезка $[a, b]$, полученное объединением $T_1$ и $T_2$ \\ \\
    В силу предыдущей леммы $\ s_f(T_1) \ \leq \ s_f(T) \ \leq \ S_f(T) \ \leq \ S_f(T_2)$
\end{proof}

\begin{definition}
    Пусть $f \colon [a, b] \to \mathbb{R}$. 
    \[J_{\ast}(f) := \sup_{T \in \mathcal{T}([a, b])}s_f(T) \ \text{ -- \textbf{нижний интеграл Дарбу}}\]
    \[J^{\ast}(f) := \inf_{T \in \mathcal{T}([a, b])}S_f(T) \ \text{ -- \textbf{верхний интеграл Дарбу}}\]
\end{definition}

\begin{lemma}
    Пусть $f \colon [a, b] \to \mathbb{R}$. Тогда \[\forall T \in \mathcal{T}([a, b]) 
    \ \hookrightarrow \ s_f(T) \leq J_{\ast} \leq J^{\ast} \leq S_f(T)\]
\end{lemma}

\begin{proof}
    Первое и последнее неравенства очевидно выполняются по определению. Докажем центральное: 
    по предыдущей лемме $s_f(T_1) \leq S_f(T_2)$. \\ \\ Возьмём инфимум по $T_2$ от обеих частей неравенства: 
    \[s_f(T_1) \leq J^{\ast}\] Возьмём супремум по $T_1$ от обеих частей неравенства: \[J_{\ast} \leq J^{\ast}\]
\end{proof}

\begin{definition}
    Пусть $f \colon [a, b] \to \mathbb{R}, \; T \in \mathcal{T}([a, b])$. Определим
    \[\Delta_f(T) := \sum_{i = 1}^{N_T} \omega_i(f)(x_i - x_{i - 1}) \text{, где}\]
    \[T = \{x_i\}^{N_T}_{i = 0} \quad \text{и} \quad \omega_i(f) \ = \sup_{x', 
    x'' \in [x_{i - 1}, x_i]} \left\lvert f(x') - f(x'') \right\rvert 
    \text{ -- колебание } f \text{ на } [x_{i - 1}, x_i]\]
\end{definition}

\begin{lemma}
    Пусть $f \colon [a, b] \to \mathbb{R}$. Тогда \[\forall T \in \mathcal{T}([a, b]) 
    \ \hookrightarrow \ \Delta_f(T) = S_f(T) - s_f(T)\]
\end{lemma}

\begin{proof}
    Если $s_f(T) = - \infty$ или $S_f(T) = + \infty$, то $S_f(T) - s_f(T) = + \infty$ \\
    Кроме того, в силу леммы 1.1 легко видеть, что в таком случае 
    \[\exists i \in \{1, \dots, N_T\}: 
    \left[
        \begin{array}{rcl}
            m_i(f) &=& - \infty \\
            M_i(f) &=& + \infty
        \end{array}
    \right.
    \implies \omega_i(f) = + \infty \implies \text{равенство тривиально выполнено}\]
    Далее $f$ ограничена на $[a, b]$. Заметим, что $\forall i \in \{1, \dots, N_T\} \ \hookrightarrow$
    \[\sup_{x', x'' \in [x_{i - 1}, x_i]} \left\lvert f(x') - f(x'') \right\rvert \ = \ M_i(f) - m_i(f)\]
    Действительно, фиксируем $x' \in [x_{i - 1}, x_i]$. Тогда 
    \[f(x'') - f(x') \ \leq \sup_{x \in [x_{i - 1}, x_i]}f(x) \ - \ f(x') \ \leq 
    \sup_{x \in [x_{i - 1}, x_i]}f(x) \ - \inf_{x \in [x_{i - 1}, x_i]}f(x)\]
    Меняя ролями $x'$ и $x''$, получим уже 
    \[\left\lvert f(x') - f(x'') \right\rvert \ \leq \ M_i(f) - m_i(f) \quad \forall x', x'' \in [x_{i - 1}, x_i]\]
    \[\implies \omega_i(f) \ \leq \ M_i(f) - m_i(f)\]
    Пойдём в другую сторону. Фиксируем $\varepsilon > 0$ и выберем $x' \in [x_{i - 1}, x_i]$ такой, что 
    \[f(x') \ > \ M_i(f) - \frac{\varepsilon}{2}\]
    Выберем $x'' \in [x_{i - 1}, x_i]$ такой, что 
    \[f(x'') \ < \ m_i(f) + \frac{\varepsilon}{2}\] \\
    \[\implies M_i(f) - m_i(f) \ \leq \ \left\lvert f(x') - f(x'') \right\rvert + 
    \frac{\varepsilon}{2} + \frac{\varepsilon}{2} \ \leq \ \omega_i(f) + \varepsilon\]
    Но $\varepsilon$ был выбран произвольно, следовательно, взяв инфимум по $\varepsilon$, 
    получаем \[M_i(f) - m_i(f) \ \leq \ \omega_i(f)\]
    Неравенство доказано в обе стороны, следовательно
    \[M_i(f) - m_i(f) \ = \ \omega_i(f) \quad \forall i \in \{1, \dots, N_T\}\]
    Вернёмся к определению $\Delta_f(T)$:
    \[\Delta_f(T) \ = \ \sum^{N_T}_{i = 1}(M_i(f) - m_i(f))(x_i - x_{i - 1}) \ = \ S_f(T) - s_f(T)\]
\end{proof}

\begin{theorem}[Первый критерий интегрируемости по Риману]
    Пусть $f \colon [a, b] \to \mathbb{R}$. Следующие условия эквивалентны: 
    \begin{enumerate}
        \item $f \in \mathcal{R}([a, b])$
        \item $\displaystyle \lim_{l(T) \to 0} \Delta_f(T) = 0$, то есть
        \[\forall \varepsilon > 0 \ \exists \delta(\varepsilon) > 0: \forall T \in 
        \mathcal{T}([a, b]): l(T) < \delta(\varepsilon) \ \hookrightarrow \ 
        \Delta_f(T) < \varepsilon\]
    \end{enumerate}
\end{theorem}

\begin{proof}
    ($1 \Longrightarrow 2$) Из интегрируемости по Риману сразу же получаем, что 
    \[\exists J \in \mathbb{R} \text{ т.ч. } \forall \varepsilon > 0 \ \exists 
    \delta(\varepsilon) > 0: \forall T \in \mathcal{T}([a, b]): l(T) < 
    \delta(\varepsilon) \ \hookrightarrow \ 
    \begin{cases}
        \left\lvert s_f(T) - J \right\rvert < \frac{\varepsilon}{2} \\
        \left\lvert S_f(T) - J \right\rvert < \frac{\varepsilon}{2} 
    \end{cases}\]
    $\implies$ по неравенству треугольника:
    \[\left\lvert s_f(T) - S_f(T) \right\rvert \ < \ \left\lvert s_f(T) - J 
    \right\rvert + \left\lvert S_f(T) - J \right\rvert \ < \ \frac{\varepsilon}{2} + 
    \frac{\varepsilon}{2} \ = \ \varepsilon\]
    $\implies$ в силу леммы 3.4 получаем 
    \[\forall T \in \mathcal{T}([a, b]): l(T) < \delta(\varepsilon) 
    \ \hookrightarrow \ \Delta_f(T) \ < \ \varepsilon\]
    \\
    ($2 \Longrightarrow 1$) Пусть 
    \[\forall \varepsilon > 0 \ \exists \delta(\varepsilon) > 0: 
    \forall T \in \mathcal{T}([a, b]): l(T) < \delta(\varepsilon) \ 
    \hookrightarrow \ \Delta_f(T) < \varepsilon\] \\
    $\implies$ в силу леммы 3.4 получаем 
    \[S_f(T) - s_f(T) < \varepsilon \quad \forall T \in \mathcal{T}([a, b]): 
    l(T) < \delta(\varepsilon)\]
    \[\implies S_f(T) < + \infty \ \text{ и } \ s_f(T) > - \infty \quad 
    \forall T \in \mathcal{T}([a, b])\]
    Тогда по лемме 1.1 получаем, что $f$ ограничена на $[a, b]$ \\ \\
    В силу леммы 3.3 \[s_f(T) \leq J_{\ast} \leq J^{\ast} \leq S_f(T) \quad 
    \forall \text{ разбиения } T\]
    \[\implies \forall T \in \mathcal{T}([a, b]): l(T) < \delta(\varepsilon) \ 
    \hookrightarrow \ J^{\ast} - J_{\ast} \leq S_f(T) - s_f(T) < \varepsilon\]
    Получаем \[0 \leq J^{\ast} - J_{\ast} < \varepsilon \quad \forall \varepsilon > 0\]
    \[\implies J^{\ast} = J_{\ast} \in \mathbb{R}\]
    Положим $J := J^{\ast} = J_{\ast}$
    \[\left\lvert s_f(T) - J \right\rvert = - s_f(T) + J_{\ast} \leq S_f(T) - 
    s_f(T) < \varepsilon \quad \forall T \in \mathcal{T}([a, b]): l(T) < 
    \delta(\varepsilon)\]
    Аналогично $\left\lvert S_f(T) - J \right\rvert < \varepsilon \quad 
    \forall T \in \mathcal{T}([a, b]): l(T) < \delta(\varepsilon)$ \\ \\
    Подведём итоги: \[\exists J \in \mathbb{R}: \ \forall \varepsilon > 0 \ 
    \exists \delta(\varepsilon) > 0: \ \forall T \in \mathcal{T}([a, b]): 
    l(T) < \delta(\varepsilon) \ \hookrightarrow \ 
    \begin{cases}
        \left\lvert s_f(T) - J \right\rvert < \varepsilon \\
        \left\lvert S_f(T) - J \right\rvert < \varepsilon
    \end{cases}\]
\end{proof}
